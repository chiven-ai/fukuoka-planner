// 精密現場日圓計算演算法 (支持化妝品/手錶 5% 與 信用卡疊加 5% 折上折)
function calculateStorePrice(storeId, originalPriceJpy, isTaxFree, isJcbOrTaishin, isTaiwaneseCard = true) {
const store = storeConfigs.find(s => s.id === storeId);
if (!store) return { discountPrice: originalPriceJpy, refund: 0, finalJpy: originalPriceJpy };
let discountPrice = originalPriceJpy;
// 1. 特店專屬折扣計算
if (store.isDaimaru) {
// 大丸百貨普通精品櫃位： 5% Guest Card 折扣 + 5% 信用卡折扣疊加 (乘積折上折)
if (isJcbOrTaishin) {
discountPrice = originalPriceJpy * (1 - store.guestDiscount) * 0.95;
} else {
discountPrice = originalPriceJpy * (1 - store.guestDiscount);
}
} else if (store.isDaimaruSpecialCampaign) {
// 化妝品/手錶專櫃：護照現折 5% + 使用台新卡(PayPay)/富邦J卡可額外再疊加 5% 專屬信用卡折扣 (折上折 5% + 5% !)
if (isTaiwaneseCard) {
const passportDiscount = originalPriceJpy * 0.95;
if (isJcbOrTaishin) {
discountPrice = passportDiscount * 0.95; // 完美雙疊加
} else {
discountPrice = passportDiscount;
}
}
} else {
// 其它特店折扣 (如 Bic Camera 7% 等)
discountPrice = originalPriceJpy * (1 - store.guestDiscount);
}
let refund = 0;
let finalJpy = discountPrice;
// 2. 退稅金額與手續費折算
if (isTaxFree && store.taxFreeRate > 0) {
const taxExclusive = discountPrice / 1.10;
const grossRefund = taxExclusive * store.taxFreeRate;
const processingFee = taxExclusive * store.taxFreeFee;
refund = grossRefund;
finalJpy = discountPrice - grossRefund + processingFee;
}
return {
discountPrice: Math.round(discountPrice),
refund: Math.round(refund),
finalJpy: Math.round(finalJpy)
};
}
// 精密信用卡回饋與限額演算法 (支援富邦 J 卡單筆千元台幣門檻判定)
function evaluateCardRebate(card, finalJpy, exRate, store) {
const twdAmount = finalJpy * exRate;
// PayPay 限制防呆
if (card.type === 'PayPay' && !store.supportsPayPay) {
return {
netSavingsTwd: -999999,
netCostTwd: 999999,
effectiveRebateRate: "0",
isNotApplicable: true,
reason: "本特店不支援 PayPay 付款"
};
}
const feeCost = twdAmount * (card.feeRate / 100);
let rawRebate = 0;
let effectivePromoRate = card.baseRate;
let isBelowThresh = false;
// 富邦 J 卡小額防呆限制：單筆未達 NT$1,000 降為基本 3.0%
if (card.id === 'fubon_j_jcb' && twdAmount < 1000) {
effectivePromoRate = 3.0; // 降為基本回饋率
isBelowThresh = true;
}
// 限額額度扣減模擬
if (card.hasCap) {
const remainingCap = Math.max(card.capAmount - card.spentAmount, 0);
if (remainingCap >= twdAmount) {
rawRebate = twdAmount * (effectivePromoRate / 100);
} else if (remainingCap > 0) {
const promoPart = remainingCap * (effectivePromoRate / 100);
const basePart = (twdAmount - remainingCap) * 1.5 / 100; // 超出上限給 1.5% 基本
rawRebate = promoPart + basePart;
} else {
rawRebate = twdAmount * 1.5 / 100;
}
} else {
rawRebate = twdAmount * (effectivePromoRate / 100);
}
let netSavingsTwd = rawRebate - feeCost;
// 3. JCB 特店與限時活動加碼補貼

ledger.splice(itemIndex, 1);
syncStorage();
updateAllCalculations();
}
